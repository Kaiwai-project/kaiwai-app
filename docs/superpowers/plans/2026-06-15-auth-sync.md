# KAIWAI 로그인 & 동기화 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 카카오/네이버/구글/이메일 4가지 로그인을 Supabase Auth로 통합하고, 즐겨찾기·피드 업로드·좋아요를 크로스 디바이스 동기화한다.

**Architecture:** 단일 `index.html`에 Supabase JS SDK CDN 추가. Email/Google은 Supabase OAuth 네이티브 사용. 카카오/네이버는 기존 SDK로 유저 정보 취득 후 shadow account(`kakao_{id}@kaiwai.app`)로 Supabase Auth에 연결. 모든 로그인은 공통 `_onAuthSuccess()` 후처리로 통일. 즐겨찾기·피드·좋아요는 Supabase DB와 localStorage 이중 저장.

**Tech Stack:** Supabase JS SDK v2 (CDN), Supabase PostgreSQL + Storage, Kakao JS SDK / Naver Login SDK (기존 유지)

### Task 1: Supabase DB + RLS + Storage 설정

**Files:** Supabase SQL Editor (브라우저, 파일 없음)

- [ ] **Step 1: SQL Editor에서 전체 실행**

  Supabase Dashboard → SQL Editor → New query → 아래 전체 붙여넣기 → Run:

  ```sql
  -- 프로필
  CREATE TABLE IF NOT EXISTS profiles (
    id          uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    nickname    text NOT NULL,
    avatar_url  text,
    provider    text NOT NULL DEFAULT 'email',
    created_at  timestamptz DEFAULT now()
  );

  -- 즐겨찾기
  CREATE TABLE IF NOT EXISTS favorites (
    id         uuid DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id    uuid REFERENCES auth.users(id) ON DELETE CASCADE,
    brand_id   integer NOT NULL,
    created_at timestamptz DEFAULT now(),
    UNIQUE(user_id, brand_id)
  );

  -- 피드 게시물
  CREATE TABLE IF NOT EXISTS feed_posts (
    id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id     uuid REFERENCES auth.users(id) ON DELETE CASCADE,
    brand_id    integer,
    image_url   text NOT NULL,
    description text DEFAULT '',
    likes_count integer DEFAULT 0,
    created_at  timestamptz DEFAULT now()
  );

  -- 피드 좋아요
  CREATE TABLE IF NOT EXISTS feed_likes (
    id         uuid DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id    uuid REFERENCES auth.users(id) ON DELETE CASCADE,
    post_id    uuid REFERENCES feed_posts(id) ON DELETE CASCADE,
    created_at timestamptz DEFAULT now(),
    UNIQUE(user_id, post_id)
  );

  -- RLS 활성화
  ALTER TABLE profiles   ENABLE ROW LEVEL SECURITY;
  ALTER TABLE favorites  ENABLE ROW LEVEL SECURITY;
  ALTER TABLE feed_posts ENABLE ROW LEVEL SECURITY;
  ALTER TABLE feed_likes ENABLE ROW LEVEL SECURITY;

  -- profiles: 본인만
  CREATE POLICY "profiles_own" ON profiles
    FOR ALL USING (auth.uid() = id) WITH CHECK (auth.uid() = id);

  -- favorites: 본인만
  CREATE POLICY "favorites_own" ON favorites
    FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

  -- feed_posts: 읽기 공개, 쓰기 본인만
  CREATE POLICY "feed_posts_read"   ON feed_posts FOR SELECT USING (true);
  CREATE POLICY "feed_posts_insert" ON feed_posts FOR INSERT WITH CHECK (auth.uid() = user_id);
  CREATE POLICY "feed_posts_update" ON feed_posts FOR UPDATE USING (auth.uid() = user_id);
  CREATE POLICY "feed_posts_delete" ON feed_posts FOR DELETE USING (auth.uid() = user_id);

  -- feed_likes: 읽기 공개, 쓰기 본인만
  CREATE POLICY "feed_likes_read"   ON feed_likes FOR SELECT USING (true);
  CREATE POLICY "feed_likes_insert" ON feed_likes FOR INSERT WITH CHECK (auth.uid() = user_id);
  CREATE POLICY "feed_likes_delete" ON feed_likes FOR DELETE USING (auth.uid() = user_id);

  -- likes_count 자동 증감 트리거
  CREATE OR REPLACE FUNCTION update_likes_count()
  RETURNS TRIGGER AS $$
  BEGIN
    IF TG_OP = 'INSERT' THEN
      UPDATE feed_posts SET likes_count = likes_count + 1 WHERE id = NEW.post_id;
    ELSIF TG_OP = 'DELETE' THEN
      UPDATE feed_posts SET likes_count = GREATEST(0, likes_count - 1) WHERE id = OLD.post_id;
    END IF;
    RETURN NULL;
  END;
  $$ LANGUAGE plpgsql SECURITY DEFINER;

  CREATE TRIGGER trg_likes_count
    AFTER INSERT OR DELETE ON feed_likes
    FOR EACH ROW EXECUTE FUNCTION update_likes_count();
  ```

  예상: `Success. No rows returned`

- [ ] **Step 2: `<script>` 최상단 (기존 `const CM = {` 바로 위)에 Supabase 초기화 블록 추가**

  ```javascript
  /* ═══════════════════════════════════
     Supabase 클라이언트
  ═══════════════════════════════════ */
  const SUPABASE_URL  = "https://xxxx.supabase.co";  // Task 0에서 메모한 값
  const SUPABASE_ANON = "eyJ...";                     // Task 0에서 메모한 anon key
  const _sb = window.supabase.createClient(SUPABASE_URL, SUPABASE_ANON);

  /* 로그인 공통 후처리 */
  async function _onAuthSuccess(session) {
    const u    = session.user;
    const meta = u.user_metadata || {};
    authUser = {
      uid:       u.id,
      email:     u.email,
      nickname:  meta.nickname || meta.full_name || u.email.split("@")[0],
      avatarUrl: meta.avatar_url || meta.picture || "",
      provider:  u.app_metadata?.provider || meta.provider || "email"
    };
    localStorage.setItem("kaiwai_user", JSON.stringify(authUser));

    await _sb.from("profiles").upsert({
      id:         authUser.uid,
      nickname:   authUser.nickname,
      avatar_url: authUser.avatarUrl,
      provider:   authUser.provider
    }, { onConflict: "id" });

    await syncFavorites();
    updateAuthBtn();
  }

  /* 소셜 shadow account 공통 로그인 */
  async function _sbSocialLogin(provider, rawId, nickname, avatarUrl) {
    const shadowEmail = `${provider}_${rawId}@kaiwai.app`;
    const shadowPw    = `kw_${provider}_${rawId}_2026`;

    const { error: signInErr } = await _sb.auth.signInWithPassword({ email: shadowEmail, password: shadowPw });

    if (signInErr?.message?.includes("Invalid login credentials")) {
      const { error: signUpErr } = await _sb.auth.signUp({
        email: shadowEmail,
        password: shadowPw,
        options: { data: { nickname, provider, avatar_url: avatarUrl || "" } }
      });
      if (signUpErr) throw signUpErr;
      await _sb.auth.signInWithPassword({ email: shadowEmail, password: shadowPw });
    } else if (signInErr) {
      throw signInErr;
    }
  }

  /* Auth 상태 변화 리스너 (Google OAuth 콜백 + 세션 갱신 처리) */
  _sb.auth.onAuthStateChange(async (event, session) => {
    if (session && (event === "SIGNED_IN" || event === "TOKEN_REFRESHED")) {
      await _onAuthSuccess(session);
      if (event === "SIGNED_IN") showToast(`${authUser.nickname}님, 환영해요! 🎀`);
    } else if (event === "SIGNED_OUT") {
      authUser = null;
      localStorage.removeItem("kaiwai_user");
      updateAuthBtn();
    }
  });

  /* 페이지 로드 시 기존 세션 복원 */
  (async () => {
    const { data: { session } } = await _sb.auth.getSession();
    if (session) await _onAuthSuccess(session);
  })();
  ```

  ```javascript
  async function doRegister() {
    const nick  = document.getElementById("regNick").value.trim();
    const email = document.getElementById("regEmail").value.trim();
    const pw    = document.getElementById("regPw").value;
    const pw2   = document.getElementById("regPw2").value;
    _hideErr("regErr");
    if (!nick || !email || !pw || !pw2) return _showErr("regErr", "모든 항목을 입력해주세요.");
    if (nick.length < 2) return _showErr("regErr", "닉네임은 2자 이상이어야 해요.");
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return _showErr("regErr", "올바른 이메일 주소를 입력해주세요.");
    if (pw.length < 8)  return _showErr("regErr", "비밀번호는 8자 이상이어야 해요.");
    if (pw !== pw2)     return _showErr("regErr", "비밀번호가 일치하지 않아요.");

    const { error } = await _sb.auth.signUp({
      email,
      password: pw,
      options: { data: { nickname: nick, provider: "email" } }
    });
    if (error) {
      const msg = error.message.includes("already registered") ? "이미 사용 중인 이메일이에요." : error.message;
      return _showErr("regErr", msg);
    }
    showToast(`${nick}님, 가입을 축하해요! 🎀`);
    closeAuth();
  }
  ```

  async function toggleFeedLike(postId, btn) {
    if (!authUser) { openAuth("login"); return; }

    const { data: existing } = await _sb
      .from("feed_likes")
      .select("id")
      .eq("user_id", authUser.uid)
      .eq("post_id", postId)
      .maybeSingle();

    const post = POSTS.find(p => p.id === postId);

    if (existing) {
      await _sb.from("feed_likes").delete().eq("id", existing.id);
      if (post) { post.liked = false; post.likes_count = Math.max(0, (post.likes_count || 1) - 1); }
    } else {
      await _sb.from("feed_likes").insert({ user_id: authUser.uid, post_id: postId });
      if (post) { post.liked = true; post.likes_count = (post.likes_count || 0) + 1; }
    }

    if (post) btn.textContent = `${post.liked ? "❤️" : "🤍"} ${post.likes_count}`;
  }
